    // Format: VNT-YYYYMMDD-XXXXX

// Statistiques d'un dÃ©part
public StockStatsDTO getStockStatistics(Long departId) {
    List<StockDepart> stocks = stockDepartRepository.findByDepartId(departId);
    
    StockStatsDTO stats = new StockStatsDTO();
    
    // QuantitÃ©s
    int totalInitial = stocks.stream()
            .mapToInt(StockDepart::getQuantiteInitiale)
            .sum();
    int totalRestant = stocks.stream()
            .mapToInt(StockDepart::getQuantiteRestante)
            .sum();
    int totalVendu = totalInitial - totalRestant;
    
    stats.setQuantiteInitiale(totalInitial);
    stats.setQuantiteRestante(totalRestant);
    stats.setQuantiteVendue(totalVendu);
    stats.setTauxUtilisation(
        totalInitial > 0 
            ? (double) totalVendu / totalInitial * 100 
            : 0
    );
    
    // Valeurs
    BigDecimal valeurInitiale = stocks.stream()
            .map(s -> s.getProduit().getPrixUnitaire()
                    .multiply(BigDecimal.valueOf(s.getQuantiteInitiale())))
            .reduce(BigDecimal.ZERO, BigDecimal::add);
    
    BigDecimal totalVentes = venteProduitRepository
            .calculateTotalByDepartId(departId);
    
    stats.setValeurStockInitial(valeurInitiale);
    stats.setTotalVentes(totalVentes);
    
    return stats;
}
